\begin{frame}{MINDSET Framework Overview}
\textbf{Multi-Regional Input-Output Model}

\vspace{0.3cm}

\textbf{Core Components:}
\begin{enumerate}
    \item \textbf{Leontief Model}: $\mathbf{q} = \mathbf{L} \times \mathbf{y}$
    \begin{itemize}
        \item $\mathbf{L} = (\mathbf{I} - \mathbf{A})^{-1}$ = Leontief inverse
        \item $\mathbf{A}$ = technical coefficients matrix
        \item Captures supply chain linkages
    \end{itemize}

    \vspace{0.2cm}

    \item \textbf{Output Changes}: $\Delta\mathbf{q} = \mathbf{L} \times \Delta\mathbf{y}$
    \begin{itemize}
        \item $\Delta\mathbf{y}$ = exogenous demand shock (investment)
        \item $\Delta\mathbf{q}$ = total output change (direct + indirect)
    \end{itemize}

    \vspace{0.2cm}

    \item \textbf{Employment Impact}: $\Delta E = \mathbf{m} \circ \Delta\mathbf{q}$
    \begin{itemize}
        \item $\mathbf{m}$ = employment multiplier vector
        \item $\circ$ = element-wise multiplication
    \end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{Employment Calculation Details}
\textbf{Employment Multiplier:}
\[
m_{i,j} = \text{EMPL\_COEF}_{i,j} \times \frac{E^{\text{base}}_{i,j}}{q^{\text{base}}_{i,j}}
\]

Where:
\begin{itemize}
    \item $i$ = country, $j$ = product
    \item $E^{\text{base}}$ = baseline employment (workers)
    \item $q^{\text{base}}$ = baseline output (\$M)
    \item EMPL\_COEF = employment elasticity parameter
\end{itemize}

\vspace{0.5cm}

\textbf{Employment Change:}
\[
\Delta E_{i,j} = m_{i,j} \times \Delta q_{i,j}
\]

\vspace{0.3cm}

\textbf{Interpretation:} Workers created per \$1M increase in output
\end{frame}

\begin{frame}{Workflow: Phase 1 - Data Preparation}
\textbf{Step 1: MRIO Aggregation}
\begin{itemize}
    \item Script: \texttt{collapse\_MRIO\_SSP.py}
    \item Aggregate 164 countries to 8 SSP regions
    \item Generate matrices: $\mathbf{Z}$, $\mathbf{Y}$, $\mathbf{VA}$, $\mathbf{E}$
    \item Output: \texttt{.pkl} and \texttt{.mat} files in \texttt{GLORIA\_db/}
\end{itemize}

\vspace{0.3cm}

\textbf{Step 2: Baseline Setup}
\begin{itemize}
    \item Script: \texttt{create\_SSP\_Y\_BASE.py}
    \item Create baseline final demand: $\mathbf{Y}^{\text{base}}$
    \item Sum household + government + investment flows
    \item Store as \texttt{Y\_BASE.mat}
\end{itemize}

\vspace{0.3cm}

\textbf{Step 3: Leontief Inverse}
\begin{itemize}
    \item Calculate $\mathbf{L} = (\mathbf{I} - \mathbf{A})^{-1}$
    \item Stored in model for repeated use
\end{itemize}
\end{frame}

\begin{frame}{Workflow: Phase 2 - Model Initialization}
\textbf{One-Time Setup (Per Batch Run):}

\vspace{0.3cm}

\begin{enumerate}
    \item \textbf{Load MRIO Data}
    \begin{itemize}
        \item Module: \texttt{exog\_vars\_SSP.py}
        \item Load aggregated matrices from \texttt{GLORIA\_db/}
        \item Load employment baseline: \texttt{LABOR\_BASE}
    \end{itemize}

    \vspace{0.2cm}

    \item \textbf{Initialize I-O Model}
    \begin{itemize}
        \item Module: \texttt{InputOutput\_SSP.py}
        \item Calculate Leontief inverse $\mathbf{L}$
        \item Store baseline output $\mathbf{q}^{\text{base}}$
    \end{itemize}

    \vspace{0.2cm}

    \item \textbf{Initialize Employment Model}
    \begin{itemize}
        \item Module: \texttt{employment.py}
        \item Calculate employment multipliers $\mathbf{m}$
        \item Ready for repeated applications
    \end{itemize}
\end{enumerate}

\vspace{0.3cm}

\textbf{Result:} All heavy computations done once, fast iterations for 469 strategies
\end{frame}

\begin{frame}{Workflow: Phase 3 - Batch Processing}
\textbf{For Each Strategy (469 iterations):}

\vspace{0.3cm}

\begin{enumerate}
    \item \textbf{Load Investment Vector}
    \begin{itemize}
        \item Read \texttt{Strategy\_*.xlsx}
        \item Map Eora26 sectors to GLORIA products
        \item Create $\Delta\mathbf{y}$ (investment demand shock)
    \end{itemize}

    \item \textbf{Calculate Output Change}
    \begin{itemize}
        \item Apply Leontief: $\Delta\mathbf{q} = \mathbf{L} \times \Delta\mathbf{y}$
        \item Returns 960-element vector (8 regions $\times$ 120 products)
    \end{itemize}

    \item \textbf{Calculate Employment}
    \begin{itemize}
        \item Apply multipliers: $\Delta E = \mathbf{m} \circ \Delta\mathbf{q}$
        \item Separate direct vs. indirect jobs
    \end{itemize}

    \item \textbf{Aggregate to ISIC Sectors}
    \begin{itemize}
        \item Map 120 GLORIA products to 20 ISIC sections (A-T)
        \item Sum employment within each sector
    \end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{Direct vs. Indirect Employment}
\textbf{Decomposition Logic:}

\vspace{0.3cm}

\textbf{Direct Employment:}
\begin{itemize}
    \item Jobs created in sectors receiving investment
    \item Condition: $\Delta y_{i,j} > 0$ (positive demand shock)
    \item Example: Construction jobs from building infrastructure
\end{itemize}

\vspace{0.5cm}

\textbf{Indirect Employment:}
\begin{itemize}
    \item Jobs created in supply chain sectors
    \item Condition: $\Delta y_{i,j} = 0$ AND $\Delta q_{i,j} > 0$
    \item Example: Steel production jobs from infrastructure construction
    \item Captures inter-industry linkages via Leontief multipliers
\end{itemize}

\vspace{0.5cm}

\textbf{Validation:}
\[
\text{Total Employment} = \text{Direct} + \text{Indirect}
\]
\end{frame}

\begin{frame}{ISIC Sector Aggregation}
\textbf{From Products to Sectors:}

\vspace{0.3cm}

120 GLORIA products $\rightarrow$ 20 ISIC Rev. 4 sections

\vspace{0.5cm}

\textbf{Example Mappings:}
\begin{itemize}
    \item \textbf{Sector A} (Agriculture): Crops, Livestock, Fishing $\rightarrow$ 8 products
    \item \textbf{Sector C} (Manufacturing): Food, Textiles, Machinery, etc. $\rightarrow$ 45 products
    \item \textbf{Sector F} (Construction): Construction services $\rightarrow$ 1 product
    \item \textbf{Sector G} (Trade): Wholesale and retail $\rightarrow$ 2 products
\end{itemize}

\vspace{0.5cm}

\textbf{Aggregation:}
\[
E^{\text{ISIC}}_s = \sum_{j \in s} \Delta E_j
\]

Where $s$ = ISIC sector, $j$ = GLORIA product
\end{frame}

\begin{frame}{Workflow: Phase 4 - Output Generation}
\textbf{Consolidation:}

\vspace{0.3cm}

\begin{itemize}
    \item Combine results from all 469 strategies
    \item Add strategy names from \texttt{ATTRIBUTE\_STRATEGY.csv}
    \item Calculate sectoral shares: $\text{share}_s = E_s / \sum_s E_s$
    \item Ensure shares sum to 1.0 per strategy
\end{itemize}

\vspace{0.5cm}

\textbf{Output File:} \texttt{employment\_consolidated.csv}

\vspace{0.3cm}

\textbf{Structure:}
\begin{itemize}
    \item Rows: $\sim$9,380 (469 strategies $\times$ 20 ISIC sectors)
    \item Columns: 10 (country, strategy, sector, jobs, shares)
    \item All 20 ISIC sectors per strategy (including zeros)
    \item 7 countries (ROW excluded)
\end{itemize}

\vspace{0.5cm}

\textbf{Location:} \texttt{SSP - Results/employment\_consolidated.csv}
\end{frame}

        \item Speeds up matrix multiplication
    \end{itemize}

\begin{frame}{Root Cause Analysis}
\textbf{Diagnostic Trace Through MINDSET Workflow:}

\vspace{0.3cm}

\textbf{Employment Multiplier Formula:}
\[
m_{i,j} = \text{EMPL\_COEF}_{i,j} \times \frac{E^{\text{base}}_{i,j}}{q^{\text{base}}_{i,j}}
\]

\vspace{0.3cm}

\textbf{⚠️ Critical Issue:} If $q^{\text{base}}$ (baseline output) is very small → multiplier becomes extremely large

\vspace{0.5cm}

\textbf{Code Location:} \texttt{SourceCode/employment.py}, lines 36--43

\vspace{0.2cm}

{\small
\texttt{empl\_multiplier = empl\_coef * (empl\_base / q\_base)}
}

\vspace{0.5cm}

\textbf{Hypothesis:} Uganda and Libya have abnormally small $q^{\text{base}}$ values in GLORIA v57 database, causing multiplier inflation.

\vspace{0.3cm}

\textbf{Why Not Other Steps?}
\begin{itemize}
    \item ✓ Investment converter logic verified correct
    \item ✓ Leontief calculations follow standard IO theory
    \item ✗ \textbf{Base data quality} is the likely culprit
\end{itemize}
\end{frame}

\begin{frame}{Investment Converter: "Strategy as Purchases" Fix}
\textbf{Conceptual Correction Already Implemented:}

\vspace{0.3cm}

\textbf{Correct Interpretation:}
\begin{quote}
Investment strategies represent \textcolor{blue}{\textbf{purchases of specific products}}, not investments made by the producing sector
\end{quote}

\vspace{0.3cm}

\textbf{Process Flow:}
\begin{enumerate}
    \item Strategy file specifies investment amounts by sector
    \item \textbf{Investment Converter} maps sector investments → product-specific final demand
    \item Uses INV\_CONV matrix: composition of investment goods
    \item Creates initial demand vector: $\Delta \mathbf{y}$
    \item Leontief model traces supply chain: $\Delta \mathbf{q} = \mathbf{L} \times \Delta \mathbf{y}$
\end{enumerate}

\vspace{0.3cm}

\textbf{Code Location:} \texttt{RunMINDSET\_EmploymentOnly\_BATCH\_FINAL.py}, lines 126--145

\vspace{0.3cm}

\textbf{Impact:}
\begin{itemize}
    \item ✓ Fixed conceptual errors in demand attribution
    \item ✓ Improved accuracy for all countries
    \item ✗ Did \textbf{not resolve} Uganda/Libya anomalies → confirms root cause is base data issue
\end{itemize}
\end{frame}

\begin{frame}{Limitations}
\textbf{Data Limitations:}
\begin{itemize}
    \item GLORIA 2019 baseline (pre-COVID)
    \item Country aggregation to SSP regions (loss of detail)
    \item Employment coefficients assumed constant
    \item \textbf{Data quality concerns:} Uganda and Libya show inflated employment estimates (20--100$\times$ higher) due to small baseline output values in GLORIA v57 → Proxy country approach recommended
\end{itemize}

\vspace{0.3cm}

\textbf{Methodological Limitations:}
